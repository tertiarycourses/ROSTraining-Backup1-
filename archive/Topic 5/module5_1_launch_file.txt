# Create the new package myworkcell_support with a dependency on myworkcell_core

cd ~/catkin_ws/src
catkin create pkg myworkcell_support --catkin-deps myworkcell_core
catkin build
source ~/catkin_ws/devel/setup.bash

# Create a directory for launch files
roscd myworkcell_support
mkdir launch

# Create a new file, workcell.launch inside the launch directory

<launch>
<node name="fake_ar_publisher" pkg="fake_ar_publisher" type="fake_ar_publisher_node" />
<node name="vision_node" pkg="myworkcell_core" type="vision_node" />
</launch>

#Test the launch file:

roslaunch myworkcell_support workcell.launch
